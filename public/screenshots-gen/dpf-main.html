/* Linux Framebuffer 数码相册 — 零图形库，全底层手写 */

// 1. Framebuffer 初始化 — mmap 显存映射
int fb_fd = open("/dev/fb0", O_RDWR);
int* fb_ptr = (int*)mmap(NULL, 800*480*4, PROT_READ|PROT_WRITE, MAP_SHARED, fb_fd, 0);

// 2. BMP 手动解析 — packed struct 读文件头
typedef struct __attribute__((packed)) {
  uint16_t type;   // "BM"
  uint32_t size;
  uint32_t offbits;   // 像素数据偏移 = 54
  uint32_t width, height;
  uint16_t bitcount;  // 24bpp
} BmpHeader;

// 3. 17 种动画切换 — 百叶窗 / 圆形收缩 / 对角线滑动
// 模式 7-8: 圆形动画 — 距离公式逐像素比较
for (int k = 0; k < 400; k += 4) {
  for (int y = 0; y < 480; y++)
    for (int x = 0; x < 800; x++)
      if ((x-400)*(x-400) + (y-240)*(y-240) < k*k)
        fb_ptr[y*800+x] = bmp_pixel[y][x];
}
